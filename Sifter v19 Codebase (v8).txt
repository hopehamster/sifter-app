- **Review**: Admins review reports within 24 hours via the admin panel, taking actions like deleting messages or banning users.
- **Audit**: All actions are logged for transparency and compliance.
This system meets App Store Guideline 1.2 and Google Play restricted content policies.

import 'package:amplify_flutter/amplify_flutter.dart';
import 'package:amplify_api/amplify_api.dart';
import 'package:flutter_background_geolocation/flutter_background_geolocation.dart';
import '../models/room_model.dart';
import '../utils/deployment_logger.dart';

class RoomService {
  Future<List<RoomModel>> getNearbyRooms(double latitude, double longitude) async {
    try {
      final result = await Amplify.API.query(
        request: GraphQLRequest<String>(
          document: '''
            query GetNearbyRooms($latitude: Float!, $longitude: Float!) {
              listRooms(filter: { 
                latitude: { between: [$latitude - 0.1, $latitude + 0.1] },
                longitude: { between: [$longitude - 0.1, $longitude + 0.1] }
              }) {
                items {
                  id
                  name
                  latitude
                  longitude
                  radius
                  creatorId
                  isNSFW
                }
              }
            }
          ''',
          variables: {'latitude': latitude, 'longitude': longitude},
        ),
      ).response;
      return result.data['listRooms']['items']
          .map((item) => RoomModel.fromJson(item))
          .toList();
    } catch (e) {
      DeploymentLogger.logError('Error getting nearby rooms', error: e);
      return [];
    }
  }

  Future<void> createRoom(RoomModel room) async {
    try {
      await Amplify.API.mutate(
        request: GraphQLRequest<String>(
          document: '''
            mutation CreateRoom($input: CreateRoomInput!) {
              createRoom(input: $input) {
                id
              }
            }
          ''',
          variables: {'input': room.toJson()},
        ),
      );
    } catch (e) {
      DeploymentLogger.logError('Error creating room', error: e);
    }
  }

  Future<void> subscribeToMessages(String roomId, Function(Map<String, dynamic>) onMessage) async {
    try {
      final subscription = Amplify.API.subscribe(
        GraphQLRequest<String>(
          document: '''
            subscription OnCreateMessage($roomId: ID!) {
              onCreateMessage(roomId: $roomId) {
                id
                content
                userId
                createdAt
              }
            }
          ''',
          variables: {'roomId': roomId},
        ),
        onEstablished: () => print('Message subscription established'),
      );
      subscription.listen((event) {
        if (event.data != null) {
          onMessage(event.data['onCreateMessage']);
        }
      });
    } catch (e) {
      DeploymentLogger.logError('Error subscribing to messages', error: e);
    }
  }

  Future<void> enforceGeofence(String roomId, double latitude, double longitude, double radius) async {
    try {
      FlutterBackgroundGeolocation.onLocation((Location location) {
        final distance = _calculateDistance(
          latitude,
          longitude,
          location.coords.latitude,
          location.coords.longitude,
        );
        if (distance > radius) {
          _deleteRoom(roomId);
        }
      });
      FlutterBackgroundGeolocation.start();
    } catch (e) {
      DeploymentLogger.logError('Error enforcing geofence', error: e);
    }
  }

  double _calculateDistance(double lat1, double lon1, double lat2, double lon2) {
    // Haversine formula to calculate distance
    const R = 6371e3; // Earth radius in meters
    final lat1Rad = lat1 * pi / 180;
    final lat2Rad = lat2 * pi / 180;
    final deltaLat = (lat2 - lat1) * pi / 180;
    final deltaLon = (lon2 - lon1) * pi / 180;

    final a = sin(deltaLat / 2) * sin(deltaLat / 2) +
        cos(lat1Rad) * cos(lat2Rad) * sin(deltaLon / 2) * sin(deltaLon / 2);
    final c = 2 * atan2(sqrt(a), sqrt(1 - a));
    return R * c;
  }

  Future<void> _deleteRoom(String roomId) async {
    try {
      await Amplify.API.mutate(
        request: GraphQLRequest<String>(
          document: '''
            mutation DeleteRoom($id: ID!) {
              deleteRoom(input: { id: $id }) {
                id
              }
            }
          ''',
          variables: {'id': roomId},
        ),
      );
    } catch (e) {
      DeploymentLogger.logError('Error deleting room', error: e);
    }
  }
}
