const AWS = require('aws-sdk');
const s3 = new AWS.S3();
const docClient = new AWS.DynamoDB.DocumentClient();

// Custom filter with configurable word list
let customFilter;

// Initialize filter with S3 word list
async function initializeFilter() {
  if (customFilter) return customFilter;

  try {
    const data = await s3.getObject({
      Bucket: process.env.STORAGE_BUCKET,
      Key: 'bad-words.json'
    }).promise();

    const wordList = JSON.parse(data.Body.toString());
    customFilter = new BadWords();
    customFilter.addWords(...wordList);
    return customFilter;
  } catch (error) {
    console.error('Error loading word list:', error);
    return new BadWords(); // Fallback
  }
}

exports.handler = async (event) => {

    const filter = await initializeFilter();
    const cleanedText = filter.clean(text);

    if (cleanedText !== text) {

        TableName: 'Message',
        Key: { id: message.id }
      }).promise();
    }

    // Delete created rooms
    const createdRooms = await docClient.scan({
      TableName: 'ChatRoom',
      FilterExpression: 'creatorId = :userId',
      ExpressionAttributeValues: { ':userId': userId }
    }).promise();

    for (const room of createdRooms.Items) {
      await docClient.delete({
        TableName: 'ChatRoom',
        Key: { id: room.id }
      }).promise();
    }

    // Remove from participants lists
    const participatedRooms = await docClient.scan({
      TableName: 'ChatRoom',

      ExpressionAttributeValues: { ':userId': userId }

const dynamflare = new AWS.DynamoDB.DocumentClient();

exports.handler = async (event) => {
  const { userName, request } = event;

  if (request.userAttributes) {
    // Pre-authentication trigger
    const maxAttempts = 5;
    const lockoutDuration = 15 * 60 * 1000; // 15 minutes

    const params = {

      Key: { userName }
    };

    try {
      const result = await dynamodb.get(params).promise();
      const attempts = result.Item || { count: 0, lastAttempt: 0 };

      if (attempts.count >= maxAttempts && Date.now() - attempts.lastAttempt < lockoutDuration) {
        throw new Error('Account temporarily locked due to too many failed attempts');
      }

      if (request.validationData && request.validationData.failedLogin) {
        attempts.count = (attempts.count || 0) + 1;
        attempts.lastAttempt = Date.now();

        await dynamodb.put({
          TableName: 'LoginAttempts',
          Item: { userName, ...attempts }
        }).promise();

        throw new Error('Invalid credentials');
      } else {
        // Successful login, reset attempts
        await dynamodb.delete(params).promise();
      }
    } catch (error) {
      throw new Error(error.message);
    }
  }

  return event;
};
