import 'package:flutter/material.dart';

class SettingsScreen extends StatelessWidget {
  Future<String> getUserId() async {
    return (await Amplify.Auth.getCurrentUser()).userId;
  }

  Future<String> getUserEmail() async {
    return (await Amplify.Auth.fetchUserAttributes())[0].value;
  }

  Future<void> resetPassword(BuildContext context) async {
    await Amplify.Auth.forgotPassword(username: await getUserEmail());
    Navigator.pushNamed(context, '/reset-password');
  }

  Future<void> deleteAccount(BuildContext context) async {
    bool confirm = await showDialog(
      context: context,
      builder: (context) => AlertDialog(
        title: Text("Delete Account"),
        content: Text("Are you sure? This cannot be undone."),
        actions: [
          TextButton(onPressed: () => Navigator.pop(context, false), child: Text("Cancel")),
          TextButton(onPressed: () => Navigator.pop(context, true), child: Text("Delete")),
        ],
      ),
    );
    if (confirm) {
      await Amplify.API.mutate(
        request: GraphQLRequest<String>(
          document: '''
            mutation DeleteAccount($userId: String!) {
              deleteAccount(userId: $userId)
            }
          ''',
          variables: {'userId': await getUserId()},
        ),
      ).response;
      await Amplify.Auth.signOut();
      Navigator.pushReplacementNamed(context, '/login');
    }
  }

  Future<void> contactSupport(BuildContext context) async {
    final url = Uri.parse("https://wa.me/+1234567890?text=Hi, I need help with Sifter!");
    if (await canLaunchUrl(url)) {
      await launchUrl(url);
    } else {
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(content: Text("Could not open WhatsApp")),
      );
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text("Settings")),
      body: Padding(
        padding: EdgeInsets.all(16),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.stretch,
          children: [
            ElevatedButton(onPressed: () => resetPassword(context), child: Text("Reset Password")),
            ElevatedButton(onPressed: () => contactSupport(context), child: Text("Contact Support")),

import 'package:flutter/material.dart';
import 'package:amplify_flutter/amplify_flutter.dart';

class ChatScreen extends StatefulWidget {
  final dynamic room;
  ChatScreen({required this.room});
  @override
  _ChatScreenState createState() => _ChatScreenState();
}

class _ChatScreenState extends State<ChatScreen> {
  String message = '';
  List messages = [];
  String userId = '';

  @override
  void initState() {
    super.initState();
    getUserId();
    subscribeToMessages();
  }

  Future<void> getUserId() async {
    userId = (await Amplify.Auth.getCurrentUser()).userId;
    setState(() {});
  }

  void subscribeToMessages() {
    Amplify.API.subscribe(
      GraphQLRequest<String>(
        document: '''
          subscription OnCreateMessage($roomId: ID!) {
            onCreateMessage(roomId: $roomId) { id content userId }
          }
        ''',
        variables: {'roomId': widget.room['id']},
      ),
      onEstablished: () => print("Subscription established"),
    ).listen((event) {
      setState(() => messages.add(event.data));
    });
  }

  Future<void> sendMessage() async {
    await Amplify.API.mutate(
      request: GraphQLRequest<String>(
        document: '''
          mutation SendMessage($roomId: ID!, $userId: String!, $content: String!) {
            sendMessage(roomId: $roomId, userId: $userId, content: $content) { id }
          }
        ''',
        variables: {
          'roomId': widget.room['id'],
          'userId': userId,
          'content': message,
        },
      ),
    ).response;
    setState(() => message = '');
  }

  Future<void> deleteMessage(String messageId) async {
    await Amplify.API.mutate(
      request: GraphQLRequest<String>(
        document: '''
          mutation DeleteMessage($roomId: ID!, $messageId: ID!, $creatorId: String!) {
            deleteMessage(roomId: $roomId, messageId: $messageId, creatorId: $creatorId)
          }
        ''',
        variables: {
          'roomId': widget.room['id'],
          'messageId': messageId,
          'creatorId': userId,
        },
      ),
    ).response;
  }

  Future<void> bootUser(String targetUserId) async {
    await Amplify.API.mutate(
      request: GraphQLRequest<String>(
        document: '''
          mutation BootUser($roomId: ID!, $userId: String!, $creatorId: String!) {
            bootUser(roomId: $roomId, userId: $userId, creatorId: $creatorId)
          }
        ''',
        variables: {
          'roomId': widget.room['id'],
          'userId': targetUserId,
          'creatorId': userId,
        },
      ),
    ).response;
  }

  Future<void> blockUser(String targetUserId) async {
    await Amplify.API.mutate(
      request: GraphQLRequest<String>(
        document: '''
          mutation BlockUser($roomId: ID!, $userId: String!, $creatorId: String!) {
            blockUser(roomId: $roomId, userId: $userId, creatorId: $creatorId)
          }
        ''',
        variables: {
          'roomId': widget.room['id'],
          'userId': targetUserId,
          'creatorId': userId,
        },
      ),
    ).response;
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text(widget.room['name'])),
      body: Column(
        children: [
          Expanded(
            child: ListView.builder(
              itemCount: messages.length,
              itemBuilder: (context, index) {
                final msg = messages[index];
                final isCreator = userId == widget.room['creatorId'];
                return ListTile(
                  title: Text(msg['content']),
                  trailing: isCreator && msg['userId'] != userId
                      ? Row(
                          mainAxisSize: MainAxisSize.min,
                          children: [
                            IconButton(
                              icon: Icon(Icons.delete),
                              onPressed: () => deleteMessage(msg['id']),
                            ),
                            PopupMenuButton(
                              itemBuilder: (context) => [
                                PopupMenuItem(value: 'boot', child: Text("Boot User")),
                                PopupMenuItem(value: 'block', child: Text("Block User")),
                              ],
                              onSelected: (value) {
                                if (value == 'boot') bootUser(msg['userId']);
                                if (value == 'block') blockUser(msg['userId']);
                              },
                            ),
                          ],
                        )
                      : null,
                );
              },
            ),
          ),
          Padding(
            padding: EdgeInsets.all(8),
            child: Row(
              children: [
                Expanded(
                  child: TextField(
                    decoration: InputDecoration(labelText: "Message"),
                    onChanged: (value) => message = value,
                  ),
                ),
                IconButton(
                  icon: Icon(Icons.send),
                  onPressed: sendMessage,
                ),
              ],
            ),
          ),
        ],
      ),
    );
  }
}

class ProfileScreen extends StatefulWidget {
  @override
  _ProfileScreenState createState() => _ProfileScreenState();
}

class _ProfileScreenState extends State<ProfileScreen> {
  String username = '';
  String email = '';

  @override
  void initState() {
    super.initState();
    fetchUserProfile();
  }

  Future<void> fetchUserProfile() async {
    final attributes = await Amplify.Auth.fetchUserAttributes();
    setState(() {
      username = attributes.firstWhere((attr) => attr.userAttributeKey == 'preferred_username', orElse: () => AuthUserAttribute(userAttributeKey: 'preferred_username', value: 'Unknown')).value;
      email = attributes.firstWhere((attr) => attr.userAttributeKey == 'email', orElse: () => AuthUserAttribute(userAttributeKey: 'email', value: 'Unknown')).value;
    });
  }

  Future<void> updateProfile(String newUsername) async {
    await Amplify.Auth.updateUserAttribute(
      userAttributeKey: 'preferred_username',
      value: newUsername,
    );
    fetchUserProfile();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text("Profile")),
      body: Padding(
        padding: EdgeInsets.all(16),
        child: Column(
          children: [
            Text("Username: $username"),
            Text("Email: $email"),
            TextField(
              decoration: InputDecoration(labelText: "New Username"),
              onSubmitted: (value) => updateProfile(value),
            ),
            ElevatedButton(
              onPressed: () => Navigator.pushNamed(context, '/settings'),
              child: Text("Go to Settings"),
            ),
          ],
        ),
      ),
    );
  }
}

import React, { useState, useEffect } from 'react';
import { API, Auth } from 'aws-amplify';

function Users() {
  const [users, setUsers] = useState([]);

  useEffect(() => {
    fetchUsers();
  }, []);

  async function fetchUsers() {
    const response = await API.graphql({
      query: `
        query ListUsers {
          listUsers { items { id email } }
        }
      `,
    });
    setUsers(response.data.listUsers.items);
  }

  async function deleteUser(userId) {
    await API.graphql({
      query: `
        mutation DeleteAccount($userId: String!) {
          deleteAccount(userId: $userId)
        }
      `,
      variables: { userId },
    });
    fetchUsers();
  }

  return (
    <div>
      <h1>User Management</h1>
      <ul>
        {users.map(user => (
          <li key={user.id}>
            {user.email}
            <button onClick={() => deleteUser(user.id)}>Delete</button>
          </li>
        ))}
      </ul>
    </div>
  );
}

        }

          listChatRooms { items { id } }
        }

      rooms: roomResponse.data.listChatRooms.items.length,
